REPORT ZRAJU_PRGM1.
TABLES MARA.

DATA : LV_MANDT TYPE MARA-MANDT,
       LV_MATNR TYPE MARA-MATNR,
       LV_ERNAM TYPE MARA-ERNAM,
       LV_VPSTA TYPE MARA-VPSTA,
       LV_PSTAT TYPE MARA-PSTAT.

PARAMETERS P_VPSTA TYPE MARA-VPSTA.

SELECT MANDT MATNR ERNAM VPSTA PSTAT INTO (LV_MANDT,LV_MATNR,LV_ERNAM,LV_VPSTA,LV_PSTAT)
FROM MARA WHERE VPSTA = P_VPSTA.

WRITE :/ LV_MANDT,LV_MATNR,LV_ERNAM,LV_VPSTA,LV_PSTAT.
ENDSELECT.
